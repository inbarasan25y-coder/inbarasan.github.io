# Batch processing with intelligent throttling

args=(row, log_writer, campaign_name)

def replace_placeholders(text: str, row_data: CsvRow) -> str:
"""Replace all placeholders with CSV column values."""
result_text = text
for column, value in row_data.items():
if column and value is not None:
clean_value = str(value).strip()
# Support multiple placeholder formats
placeholders = [
f'[{column}]',
f'{{{column}}}',
f'{{{{{column}}}}}'
]
for placeholder in placeholders:
result_text = result_text.replace(placeholder, clean_value)
return result_text
